# MONETIZATION SYSTEM IMPLEMENTATION FOR GRADUS MEDIA

## PROJECT CONTEXT
You are implementing a freemium-to-paid monetization system for Gradus Media, a Ukrainian HoReCa (Hotels, Restaurants, Cafes) industry AI consulting platform. The platform features "Alex Gradus," an AI chatbot that provides industry expertise.

**Current Tech Stack:**
- Frontend: React + TypeScript + Vite (Cloudflare Pages)
- Backend: FastAPI + Python (Render.com at gradus-ai.onrender.com)
- Database: PostgreSQL (Neon)
- Payment Gateway: LiqPay (Ukrainian payment processor)

**Current State:**
- Chat interface exists at `/chat` with basic 5-question free limit
- Homepage exists with hero section and news grid
- Backend has basic Maya chat API
- No email collection, no pricing page, no payment integration

**Business Model:**
- FREE: 5 questions/day per email
- STANDARD: $7/month - unlimited questions + weekly reports + supplier database
- PREMIUM: $10/month - everything + monthly video consultation + exclusive discounts

-- Maya Users Table
CREATE TABLE IF NOT EXISTS maya_users (
    email VARCHAR(255) PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    position VARCHAR(100),
    registered_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    questions_today INT DEFAULT 0,
    questions_limit INT DEFAULT 5,
    last_question_at TIMESTAMP,
    last_reset_date DATE,
    subscription_tier VARCHAR(50) DEFAULT 'free',
    subscription_expires_at TIMESTAMP,
    liqpay_order_id VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Index for faster lookups
CREATE INDEX IF NOT EXISTS idx_maya_users_email ON maya_users(email);
CREATE INDEX IF NOT EXISTS idx_maya_users_subscription ON maya_users(subscription_tier, subscription_expires_at);

-- Subscription Transactions Table (for payment tracking)
CREATE TABLE IF NOT EXISTS maya_subscriptions (
    id SERIAL PRIMARY KEY,
    email VARCHAR(255) REFERENCES maya_users(email),
    tier VARCHAR(50) NOT NULL,
    billing_cycle VARCHAR(20) NOT NULL,
    amount DECIMAL(10, 2) NOT NULL,
    currency VARCHAR(3) DEFAULT 'USD',
    liqpay_order_id VARCHAR(255) UNIQUE,
    payment_status VARCHAR(50) DEFAULT 'pending',
    started_at TIMESTAMP,
    expires_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX IF NOT EXISTS idx_subscriptions_email ON maya_subscriptions(email);
CREATE INDEX IF NOT EXISTS idx_subscriptions_order ON maya_subscriptions(liqpay_order_id);

from fastapi import APIRouter, HTTPException, Request
from pydantic import BaseModel, EmailStr
from datetime import datetime, timedelta, date
import logging

router = APIRouter(prefix="/api/maya", tags=["maya"])
logger = logging.getLogger(__name__)

class UserRegistration(BaseModel):
    name: str
    email: EmailStr
    position: str

class ChatRequest(BaseModel):
    message: str
    email: EmailStr

class ChatResponse(BaseModel):
    reply: str
    remaining_questions: int

@router.post("/register")
async def register_user(user: UserRegistration, db):
    """Register new user and initialize their free trial"""
    try:
        # Check if user already exists
        existing = await db.fetchrow(
            "SELECT email FROM maya_users WHERE email = $1",
            user.email
        )
        
        if existing:
            # Return existing user's remaining questions
            user_data = await db.fetchrow("""
                SELECT questions_today, questions_limit, last_reset_date
                FROM maya_users WHERE email = $1
            """, user.email)
            
            # Reset counter if new day
            today = date.today()
            if user_data["last_reset_date"] != today:
                await db.execute("""
                    UPDATE maya_users 
                    SET questions_today = 0, last_reset_date = $1
                    WHERE email = $2
                """, today, user.email)
                remaining = user_data["questions_limit"]
            else:
                remaining = user_data["questions_limit"] - user_data["questions_today"]
            
            return {
                "success": True,
                "existing_user": True,
                "remaining_questions": remaining
            }
        
        # Insert new user
        await db.execute("""
            INSERT INTO maya_users 
            (email, name, position, questions_today, questions_limit, last_reset_date)
            VALUES ($1, $2, $3, 0, 5, $4)
        """, user.email, user.name, user.position, date.today())
        
        logger.info(f"New user registered: {user.email}")
        
        return {
            "success": True,
            "existing_user": False,
            "remaining_questions": 5
        }
        
    except Exception as e:
        logger.error(f"Registration error: {e}")
        raise HTTPException(status_code=500, detail=str(e))


@router.post("/chat", response_model=ChatResponse)
async def chat(request: ChatRequest, db):
